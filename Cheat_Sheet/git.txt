-----------
git commend
-----------

[rebase without changing local changes]
git pull --rebase --autostash

[push all tags from your local to the remote]
git push --tags

[delete remote tag]
git push --delete origin <tag_name>  

[view the history of a single file]
git log -p <file_name>

[drop local change for one single file]
git checkout HEAD -- <file_name>

[edit commit message]
git commit --amend

[compare two files]
git diff --no-index <file_name_1> <file_name_2>

[undo add]
git reset

[check origin url]
git remote -v

[update origin url]
git remote set-url origin <new_url>

[enable long file_name in Windows] (Remember to open Git Bash with administrator permission)
git config --system core.longpaths true

[remove all histories]
rm -rf .git
git init
git add .
git commit -m "first_push"
git remote add origin https://github.com/blacksnail789521/Sharing-Files.git
git push -u --force origin master
exit
